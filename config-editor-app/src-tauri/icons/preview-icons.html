1. Install converter:

brew install librsvg

2. Convert your favorite to PNG:

cd src-tauri/icons

# Choose one:
rsvg-convert -w 512 -h 512 icon-modern-v1.svg > icon.png  # Purple gradient
rsvg-convert -w 512 -h 512 icon-modern-v2.svg > icon.png  # Dark sleek
rsvg-convert -w 512 -h 512 icon-modern-v3.svg > icon.png  # Indigo minimal
rsvg-convert -w 512 -h 512 icon-modern-v4.svg > icon.png  # Elegant gray

3. Generate all sizes:

SOURCE="icon-modern-v1.svg"  # Change to your chosen version

rsvg-convert -w 32 -h 32 $SOURCE > 32x32.png
rsvg-convert -w 128 -h 128 $SOURCE > 128x128.png
rsvg-convert -w 256 -h 256 $SOURCE > 128x128@2x.png

# macOS .icns
mkdir icon.iconset
sips -z 16 16 icon.png --out icon.iconset/icon_16x16.png
sips -z 32 32 icon.png --out icon.iconset/icon_16x16@2x.png
sips -z 32 32 icon.png --out icon.iconset/icon_32x32.png
sips -z 64 64 icon.png --out icon.iconset/icon_32x32@2x.png
sips -z 128 128 icon.png --out icon.iconset/icon_128x128.png
sips -z 256 256 icon.png --out icon.iconset/icon_128x128@2x.png
sips -z 256 256 icon.png --out icon.iconset/icon_256x256.png
sips -z 512 512 icon.png --out icon.iconset/icon_256x256@2x.png
sips -z 512 512 icon.png --out icon.iconset/icon_512x512.png
cp icon.png icon.iconset/icon_512x512@2x.png
iconutil -c icns icon.iconset
rm -rf icon.iconset
cp icon.png icon.ico

cd ../..

4. Run the app:

npm run tauri:dev
